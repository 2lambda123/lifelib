    [docs]def NetPremRate(basis):
    """Net Premium Rate"""

    gamma2 = LoadMaintSA2
    comf = LifeTable[Sex, IntRate(basis), TableID(basis)]

    if Product == 1 or Product == 2:
        return (comf.Axn(x, n) + gamma2 * comf.AnnDuenx(x, n - m, 1, m)) / comf.AnnDuenx(x, n)

    elif Product == 3:
        return (comf.Axn(x, n) + gamma2 * comf.AnnDuenx(x, n - m, 1, m)) / comf.AnnDuenx(x, n)

    else:
        raise ValueError('invalid product')
    



    [docs]def GrossPremRate():
    """Gross Premium Rate  per Sum Assured"""

    alpha = LoadAcqSA
    beta = LoadMaintGP
    gamma = LoadMaintSA
    gamma2 = LoadMaintSA2
    delta = LoadMaintPremWaiverGP

    comf = LifeTable[Sex, IntRate('PREM'), TableID('PREM')]

    if Product == 1 or Product == 2:
        return (comf.Axn(x, n) + alpha + gamma * comf.AnnDuenx(x, n, 12)
                + gamma2 * comf.AnnDuenx(x, n - m, 1, m)) / (1 - beta - delta) / 12 / comf.AnnDuenx(x, m, 12)

    elif Product == 3:
        return (comf.Exn(x, n) + comf.Axn(x, n) + alpha + gamma * comf.AnnDuenx(x, n, 12)
                + gamma2 * comf.AnnDuenx(x, n - m, 1, m)) / (1 - beta - delta) / 12 / comf.AnnDuenx(x, m, 12)
    else:
        raise ValueError('invalid product')

    [docs]def NLPReserveRate(basis, t):
    """Net level premium reserve rate"""

    gamma2 = LoadMaintSA2

    lt = LifeTable[Sex, IntRate(basis), TableID(basis)]

    if t <= m:
        return lt.Axn(x + t, n - t) + gamma2 * lt.AnnDuenx(x + t, n - m, 1, m - t) \
                - NetPremRate(basis) * lt.AnnDuenx(x + t, m - t)
    else:
        return lt.Axn(x + t, n - t) + gamma2 * lt.AnnDuenx(x + t, n - m, 1, m - t)
